G = -exp(i k R) / (4 pi R)
check(d(G,R) == G (i k - 1/R))

G = -exp(-i k R) / (4 pi R)
check(d(G,R) == G (-i k - 1/R))

G = -exp(i k R) / (4 pi R)
f = 4 pi R^2 d(G,R)
f
check(eval(f,R,0) == 1)

G = -exp(-i k R) / (4 pi R)
f = 4 pi R^2 d(G,R)
f

G = -exp(i k R) / (4 pi R)

f

G = -exp(-i k R) / (4 pi R)

f
