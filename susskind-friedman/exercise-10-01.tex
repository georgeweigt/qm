\fbox{\parbox{\dimexpr\linewidth-2\fboxsep-2\fboxrule}{
Exercise 10.1.
Find the second time derivative of $x$ in Eq.~10.9,
and thereby show that it solves Eq.~10.8.
}}

\bigskip
We are given
\begin{equation*}
-\omega^2x=\ddot x\tag{10.8}
\end{equation*}

and
\begin{equation*}
x=A\cos(\omega t)+B\sin(\omega t)\tag{10.9}
\end{equation*}

Hence
\begin{equation*}
\ddot x=\frac{d^2}{dt^2}\left(A\cos(\omega t)+B\sin(\omega t)\right)
=-A\omega^2\cos(\omega t)-B\omega^2\sin(\omega t)=-\omega^2 x
\end{equation*}
