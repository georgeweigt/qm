\FBOX{
11.30
In this problem we will recover the results Section 11.2.1 directly from the
Hamiltonian for a charged particle in an electromagnetic field (Equation 4.188). An
electromagnetic wave can be described by the potentials
\begin{equation*}
\mathbf A=\frac{\mathbf E_0}{\omega}\sin(\mathbf k\cdot\mathbf r-\omega t),
\qquad\varphi=0
\end{equation*}

where in order to satisfy Maxwell’s equations, the wave must be transverse
($\mathbf E_0\cdot\mathbf k=0$)
and of course travel at the speed of light ($\omega=c|\mathbf k|$).

(a) Find the electric and magnetic fields for this plane wave.

(b) The Hamiltonian may be written as $H^0+H'$ where $H^0$
is the Hamiltonian in the
absence of the electromagnetic wave and $H'$ is the perturbation.
Show that the
perturbation is given by
\begin{equation*}
\hat H'(t)=\frac{e}{2im\omega}e^{i\mathbf k\cdot\mathbf r}
\mathbf E_0\cdot\hat{\mathbf p}e^{-i\omega t}
-\frac{e}{2im\omega}e^{-i\mathbf k\cdot\mathbf r}
\mathbf E_0\cdot\hat{\mathbf p}e^{i\omega t},
\tag{11.27}
\end{equation*}

plus a term proportional to $E_0^2$ that we will ignore.
{\it Note}: the first term corresponds
to absorption and the second to emission.

(c) In the dipole approximation we set $e^{i\mathbf k\cdot\mathbf r}\approx1$.
With the electromagnetic wave
polarized along the $z$ direction, show that the matrix element for absorption is then
\begin{equation*}
V_{ba}=-\frac{\omega_0}{\omega}\wp E_0.
\end{equation*}

Compare Equation 11.41. They’re not {\it exactly} the same; would the difference effect
our calculations in Section 11.2.3 or 11.3?
Why or why not? {\it Hint}: To turn the
matrix element of $\mathbf p$ into a matrix element of $\mathbf r$, you need to prove the following
identity: $im[\hat H^0,\hat{\mathbf r}]=\hbar\hat{\mathbf p}$.
}

(a)
\begin{align*}
\mathbf E&=-\nabla\phi-\frac{\partial}{\partial t}\mathbf A
=\mathbf E_0\cos(\mathbf k\cdot\mathbf r-\omega t)
\\[1ex]
\mathbf B&=\nabla\times\mathbf A
=\mathbf k\times\frac{\mathbf E_0}{\omega}\cos(\mathbf k\cdot\mathbf r-\omega t)
=\mathbf k\times\frac{\mathbf E}{\omega}
\end{align*}

(b) Recall that $e=-q$. Electron charge $q$ is negative hence $e$ is positive.
\begin{align*}
H&=\frac{1}{2m}(\mathbf p+e\mathbf A)\cdot(\mathbf p+e\mathbf A)-e\varphi+V
\\[1ex]
&=\frac{1}{2m}\left(\mathbf p^2+\mathbf p\cdot e\mathbf A+e\mathbf A\cdot\mathbf p+e^2\mathbf A^2\right)
-e\varphi+V
\end{align*}

Let
\begin{equation*}
H^0=\frac{1}{2m}\mathbf p^2-e\varphi+V
\end{equation*}

and
\begin{equation*}
H'=\frac{1}{2m}\left(\mathbf p\cdot e\mathbf A+e\mathbf A\cdot\mathbf p+e^2\mathbf A^2\right)
\end{equation*}

Note that for any wave function $\psi(\mathbf r,t)$
\begin{equation*}
(\mathbf p\cdot e\mathbf A+e\mathbf A\cdot\mathbf p)\psi(\mathbf r,t)
=2e\mathbf A\cdot\mathbf p\psi(\mathbf r,t)-i\hbar\psi(\mathbf r,t)\nabla\cdot\mathbf A
\end{equation*}

Hence for the Coulomb guage $\nabla\cdot\mathbf A=0$ we have the identity
\begin{equation*}
\mathbf p\cdot e\mathbf A+e\mathbf A\cdot\mathbf p=2e\mathbf A\cdot\mathbf p
\end{equation*}

Discard the $e^2\mathbf A^2$ term to obtain
\begin{equation*}
H'=\frac{e}{m}\mathbf A\cdot\mathbf p
=\frac{e}{m\omega}\sin(\mathbf k\cdot\mathbf r-\omega t)\mathbf E_0\cdot\mathbf p
\end{equation*}

In exponential form
\begin{equation*}
H'=\frac{e}{2im\omega}\exp(i\mathbf k\cdot\mathbf r-i\omega t)\mathbf E_0\cdot\mathbf p
-\frac{e}{2im\omega}\exp(-i\mathbf k\cdot\mathbf r+i\omega t)\mathbf E_0\cdot\mathbf p
\end{equation*}

Noting that $e^{i\omega t}$ is independent of $\mathbf r$ we can also write
\begin{equation*}
H'=\frac{e}{2im\omega}\exp(i\mathbf k\cdot\mathbf r)\mathbf E_0\cdot\mathbf pe^{-i\omega t}
-\frac{e}{2im\omega}\exp(-i\mathbf k\cdot\mathbf r)\mathbf E_0\cdot\mathbf pe^{i\omega t}
\end{equation*}

(c) The matrix element for absorption is
\begin{equation*}
\langle\psi_b|H'_{ba}|\psi_a\rangle
=\frac{e}{2im\omega}\langle\psi_b|\mathbf E_0\cdot\mathbf p|\psi_a\rangle e^{-i\omega t}
\end{equation*}

For electromagnetic waves
polarized along the $z$ direction
\begin{equation*}
\mathbf E_0\cdot\mathbf p
=E_0\begin{pmatrix}0\\0\\1\end{pmatrix}
\cdot\begin{pmatrix}p_x\\p_y\\p_z\end{pmatrix}
=E_0p_z
\end{equation*}

Hence
\begin{equation*}
\langle\psi_b|H'_{ba}|\psi_a\rangle
=\frac{eE_0}{2im\omega}\langle\psi_b|p_z|\psi_a\rangle e^{-i\omega t}
\end{equation*}

Then by
\begin{equation*}
p_z=\frac{im}{\hbar}[H^0,z]
=\frac{im}{\hbar}(H^0z-zH^0)
\end{equation*}

we have
\begin{align*}
\langle\psi_b|H'_{ba}|\psi_a\rangle
&=\frac{eE_0}{2\omega\hbar}
\left[\langle\psi_b|H^0z|\psi_a\rangle-\langle\psi_b|zH^0|\psi_a\rangle\right]
e^{-i\omega t}
\\[1ex]
&=\frac{eE_0}{2\omega\hbar}
\left[\langle\psi_b|E_bz|\psi_a\rangle-\langle\psi_b|zE_a|\psi_a\rangle\right]
e^{-i\omega t}
\\[1ex]
&=\frac{eE_0}{2\omega}\frac{E_b-E_a}{\hbar}
\langle\psi_b|z|\psi_a\rangle
e^{-i\omega t}
\\[1ex]
&=\frac{eE_0}{2}\frac{\omega_0}{\omega}
\langle\psi_b|z|\psi_a\rangle
e^{-i\omega t}
\end{align*}

Hence
\begin{equation*}
V_{ba}=2e^{i\omega t}\langle\psi_b|H'_{ba}|\psi_a\rangle
=eE_0\frac{\omega_0}{\omega}
\langle\psi_b|z|\psi_a\rangle
\end{equation*}

Rearrange as
\begin{equation*}
V_{ba}=-\frac{\omega_0}{\omega}\wp E_0
\end{equation*}

where
\begin{equation*}
\wp=-e\langle\psi_b|z|\psi_a\rangle
\end{equation*}
