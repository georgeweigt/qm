\fbox{\parbox{\dimexpr\linewidth-2\fboxsep-2\fboxrule}{
19-1.
Find formulas for the radius $r$, the velocity $v$, and the momentum
$p=mv$ for the circular Bohr orbit for each $n=1,2,3,4,\ldots$
Compare $rp$ with $\hbar/2$ for each $n$.
You do not need to use numbers for that.
Write your answers in terms of $n$, $\hbar$, $m$, and $Ze^2$ only.
}}

\bigskip
Start with
\begin{equation*}
\varepsilon_n=-\frac{m(Ze^2)^2}{2\hbar^2n^2}
\end{equation*}

and
\begin{equation*}
\varepsilon=-\frac{1}{2}\frac{Ze^2}{r}
\end{equation*}

Hence
\begin{equation*}
r=-\frac{1}{2}\frac{Ze^2}{\varepsilon}=\frac{\hbar^2n^2}{mZe^2}\tag{1}
\end{equation*}

By the formula
\begin{equation*}
mv^2=\frac{Ze^2}{r}
\end{equation*}

we have
\begin{equation*}
v=\frac{Ze^2}{\hbar n}\tag{2}
\end{equation*}

For $p=mv$ we have
\begin{equation*}
p=\frac{mZe^2}{\hbar n}\tag{3}
\end{equation*}

For $rp$ we have
\begin{equation*}
rp=\hbar n\tag{4}
\end{equation*}
