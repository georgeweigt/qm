\fbox{\parbox{\dimexpr\linewidth-2\fboxsep-2\fboxrule}{
27-3.
Show that for two particles
\begin{align*}
\hbar J_1&=R_2\hbar K_3-R_3\hbar K_2+Q_2P_3-Q_3P_2,
\\
\hbar J_2&=R_3\hbar K_1-R_1\hbar K_3+Q_3P_1-Q_1P_3,
\\
\hbar J_3&=R_1\hbar K_2-R_2\hbar K_1+Q_1P_2-Q_2P_1,
\end{align*}
}}

\bigskip
We have
\begin{equation*}
\hbar J_1=L_{e1}+L_{n1}=Q_{e2}P_{e3}-Q_{e3}P_{e2}+Q_{n2}P_{n3}-Q_{n3}P_{n2}
\end{equation*}

From problem 27-1
\begin{align*}
Q_{e2}&=R_2+\frac{m_n}{m_n+m_e}Q_2 & Q_{e3}&=R_3+\frac{m_n}{m_n+m_e}Q_3
\\
Q_{n2}&=R_2-\frac{m_e}{m_n+m_e}Q_2 & Q_{n3}&=R_3-\frac{m_e}{m_n+m_e}Q_3
\end{align*}

and
\begin{align*}
P_{e2}&=\frac{m_e}{m_n+m_e}\hbar K_2+P_2 & P_{e3}&=\frac{m_e}{m_n+m_e}\hbar K_3+P_3
\\
P_{n2}&=\frac{m_n}{m_n+m_e}\hbar K_2-P_2 & P_{n3}&=\frac{m_n}{m_n+m_e}\hbar K_3-P_3
\end{align*}
